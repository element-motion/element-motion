# CircleExpand

Will animate a circle from the origin element to cover the entire viewport,
and then fade out.

Generally you should use [CircleExpand](/circle-expand) and [CricleShrink](/circle-shrink) together to seamlessly transition the background between pages.

## Usage

```js
import { CircleExpand } from '@element-motion/core';
```

<Playground>
  <Common.Toggler>
    {({ shown, toggle }) =>
      !shown ? (
        <Styled.Container>
          <Motion name="circle-expand" key="1">
            <CircleExpand background="#ff5e6d">
              {({ ref, style }) => (
                <Styled.Button onClick={() => toggle()} style={style} ref={ref} />
              )}
            </CircleExpand>
          </Motion>
        </Styled.Container>
      ) : (
        <Styled.Container onClick={() => toggle()} interactive>
          <Motion name="circle-expand" key="2">
            {({ ref, style }) => <div style={style} ref={ref} />}
          </Motion>
        </Styled.Container>
      )
    }
  </Common.Toggler>
</Playground>
